    [HttpGet]
    [Route("/search_student_by_identity/{identidade}")]
    public async Task<ActionResult<IEnumerable<DBStudentsModels>>> GetStudentsByIdentity(string identidade)
    {
        
        var student = await _context.DBStudents.FirstOrDefaultAsync(d => d.Identidade == identidade);

        if (student == null)
        {
            return NotFound();

        }
        return Ok(student);    

    }

    [HttpPost]
    [Route("/addstudent")]
    public async Task<ActionResult<DBStudentsModels>> PostStudent([FromBody]DBStudentsModels new_student)
    {
        _context.DBStudents.Add(new_student);
        await _context.SaveChangesAsync();
        
        return CreatedAtAction("GetStudents", new { id = new_student.Id}, new_student);
    }
